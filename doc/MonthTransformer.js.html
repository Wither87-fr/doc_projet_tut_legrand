/**
 * Récupére le numéro d'un mois et retourne son écriture courte (1 => jan., 2 => fev., ... 12 => dec.)
 * @param {number} int le numéro du mois
 * @returns le nom court du mois
 */
function turnIntToMonthString(int) {
	switch (int) {
		case 1:
			return "jan.";
		case 2:
			return "fev.";
		case 3:
			return "mars";
		case 4:
			return "avr.";
		case 5:
			return "mai";
		case 6:
			return "juin";
		case 7:
			return "juil.";
		case 8:
			return "aout";
		case 9:
			return 'sept.';
		case 10:
			return "oct.";
		case 11:
			return "nov.";
		case 12:
			return "dec.";
		default:
			return "unknown";
	}
}

/**
 * Calcule l'objectif d'un mois en prenant en compte les différence de jours
 * @param {number} numMonth le numéro du mois pour lequel calculer
 * @param {number} dailyObjective l'objectif journalier
 * @returns la multiplication de l'objectif par le nombre de mois
 */
function getObjectifByMonth(numMonth, dailyObjective) {
	switch (numMonth) {
		case 1:
		case 3:
		case 5:
		case 7:
		case 8:
		case 10:
		case 12:
			return dailyObjective * 31;
		case 2:
			return dailyObjective * fevrierDays();
		case 4:
		case 6:
		case 9:
		case 11:
			return dailyObjective * 30;
		default:
			return dailyObjective;
	}
}

/**
 * Permet de connaitre le nombre de jours dans le mois de février de l'année courrante
 * @returns Le nombre de jours de février de l'année en cours
 */
function fevrierDays() {
	let year = new Date().getFullYear();
	if (isBisextile(year)) {
		return 29;
	} else {
		return 28;
	}
}

/**
 * 
 * @param {number} year l'annéee 
 * @returns true si l'année est bisextile, false sinon
 */
function isBisextile(year) {
	return (year % 4 === 0 && year % 100 > 0) || (year % 400 === 0)
}
